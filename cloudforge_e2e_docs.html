Final Working Jenkinsfile
✅ Working
▼
groovy — Jenkinsfile 
pipeline {
    agent any
    tools {
        jdk 'jdk21'
        maven 'maven3'
    }
    environment {
        IMAGE_NAME = "jayanchule0699/bankapp"
        IMAGE_TAG  = "${BUILD_NUMBER}"
    }
    stages {
        stage('Check Java Version') {
            steps {
                sh '''
                echo "JAVA_HOME=$JAVA_HOME"
                java -version && javac -version && mvn -version
                '''
            }
        }
        stage('Git Checkout') {
            steps {
                git branch: 'main',
                    credentialsId: 'git',
                    url: 'https://github.com/Jayprakash-A06/mega_project_ci.git'
            }
        }
        stage('Compile') { steps { sh "mvn compile" } }
        stage('Testing') { steps { sh "mvn test" } }
        stage('Trivy FS Scan') {
            steps {
                sh '''
                trivy fs --exit-code 0 --format table -o fs-report.txt .
                echo "Trivy Exit Code = $?"
                '''
            }
        }
        stage('Sonar Analysis') {
            steps {
                withSonarQubeEnv('sonar') {
                    sh '''mvn clean verify sonar:sonar \
                      -Dsonar.projectKey=gcbank \
                      -Dsonar.projectName=gcbank \
                      -Dsonar.java.binaries=target'''
                }
            }
        }
        stage('Quality Gate Check') {
            steps {
                timeout(time: 1, unit: 'HOURS') {
                    waitForQualityGate abortPipeline: false,
                    credentialsId: 'sonar-token'
                }
            }
        }
        stage('Build') { steps { sh "mvn package" } }
        stage('Publish To Nexus') {
            steps {
                withMaven(globalMavenSettingsConfig: 'devopsshack',
                          jdk: '', maven: 'maven3', mavenSettingsConfig: '') {
                    sh "mvn deploy"
                }
            }
        }
        stage('Docker Image Build & Tag') {
            steps {
                script {
                    withDockerRegistry(credentialsId: 'docker-hub-cred', url: '') {
                        sh "docker build -t ${IMAGE_NAME}:${IMAGE_TAG} ."
                        sh "docker tag ${IMAGE_NAME}:${IMAGE_TAG} ${IMAGE_NAME}:latest"
                    }
                }
            }
        }
        stage('Trivy Image Scan') {
            steps {
                sh """
                export TMPDIR=/var/lib/jenkins/trivy-tmp
                trivy image \
                  --exit-code 0 --format table \
                  --cache-dir /var/lib/jenkins/trivy-cache \
                  --scanners vuln \
                  -o image-report.txt \
                  ${IMAGE_NAME}:${IMAGE_TAG}
                """
            }
        }
        stage('Docker Image Push') {
            steps {
                script {
                    withDockerRegistry(credentialsId: 'docker-hub-cred', url: '') {
                        sh "docker push ${IMAGE_NAME}:${IMAGE_TAG}"
                        sh "docker push ${IMAGE_NAME}:latest"
                    }
                }
            }
        }
    }
}
